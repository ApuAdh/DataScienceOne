Well there are hundreds of pokemon for every type so I decided to use the following code to filter out the strongest and the fastest pokemon for the given type:

def strongest_pokemon(type, OffenceOrDefence): filtered_df = df[(df["Type1"] == type) | (df["Type2"] == type)] filtered_df = filtered_df.assign( Preferred_Attack = filtered_df[['Attack', 'Special_Attack']].max(axis=1) ) if OffenceOrDefence.lower() == "offence": offence_filter = filtered_df.query("Preferred_Attack >= 110 and Speed >=110") graph = sns.pointplot(data=offence_filter, x="Preferred_Attack", y="Speed", hue="Name") sns.move_legend(graph, "upper left", bbox_to_anchor=(1, 1)) if OffenceOrDefence.lower() == "defence": defence_filter = filtered_df.query("Defence >= 100 and Special_Defence >=100") graph = sns.pointplot(data=defence_filter, x="Defence", y="Special_Defence", hue="Name") sns.move_legend(graph, "upper left", bbox_to_anchor=(1, 1))
